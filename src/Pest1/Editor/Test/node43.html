Neue Abstrakte Syntax

  

Statechart      ::      events          : Events        /* all events */
                        bvars           : Bvars         /* all bvars */
                        names           : Paths         /* min. path suffixes */
                        state           : Or_State      /* Rootstate */

SubSyn          =       State 
                        | Tr 
                        | Connector
                        parent          : State

Statenames      =       Str -set
Statename       =       Str

Paths           =       Pathname -set
Path            =       Str *

Bvars           =       Bvar -set
Bvar            =       Str

States          =       State -set

State           =       AND-State 
                        | OR-State 
                        | Basic-State
                        name : String

AND-State       ::      substates       : States

OR-State        ::      substates       : States
                        trs             : Trs
                        conns           : Connectors

BASIC-States    ::      default         : Boolean

Ref_State       =       Basic_State

Connectors      =       Connector -set

Connector       =       name            : Conname

Conname         =       Str

Trs             =       Tr -set

Tr              ::      source          : TrAnchor
                        target          : TrAnchor
                        tlabel          : Label

Label           ::      guard        : Guard
                        action       : Action

TrAnchor        =       State 
                        | Connector 
                        | UNDEFINED

Guard           =       emptyexpr     : Dummy
                        | event       : Event
                        | bvar        : Bvar
                        | negguard    : Guard
                        | compp       : Comppath
                        | compg       : Compguard
                        | undet       : Str

Comppath        ::      s_pop   : Pathop
                        s_ppath : Path

Pathop          ::      IN 
                        | ENTERED 
                        | EXITED

Compguard       ::      s_eop   : Op
                        s_elhs  : Guard
                        s_erhs  : Guard

Op              =       ANDOP 
                        | OROP 
                        | IMPLIES 
                        | EQUIV

Action          =       s_block         : Aseq
                        | s_egen        : Event
                        | s_bstmt       : Boolstmt
                        | s_empty       : Dummy

Aseq            =       Action *

Boolstmt        =       mtrue         : Bvar
                        | mfalse      : Bvar
                        | bass        : Bassign

Bassign         ::      blhs          : Bvar
                        brhs          : Cond

Dummy           =       Str
